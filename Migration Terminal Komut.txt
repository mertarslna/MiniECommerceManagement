dotnet ef database update <ONCEKI_MIGRATION_ADI> --project MiniECommerce.DataAccess --startup-project MiniECommerce.API
dotnet ef migrations remove --project MiniECommerce.DataAccess --startup-	project MiniECommerce.API

// Remove işlemi - Migration Sadece Oluşturulduysa (Henüz Update Yapılmadıysa)
